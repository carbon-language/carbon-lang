; RUN: | FileCheck -match-full-lines %s -check-prefixes=DAG,DAG-Q \

; RUN: | FileCheck -match-full-lines %s -check-prefixes=DAG,DAG-V,DAG-VQ \

; RUN:             -implicit-check-not='remark:'
