  find_library(LIBPGMATH pgmath PATHS ${LIBPGMATH_DIR})

    add_compile_definitions(LINK_WITH_LIBPGMATH)
    message(STATUS "Found libpgmath: ${LIBPGMATH}")
  else()
    message(STATUS "Libpgmath not found in: ${LIBPGMATH_DIR}")
  endif()
endif()
